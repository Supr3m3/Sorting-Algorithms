                                    void someSort(int arr, int length) {
                                      int j, temp;
                                      for (int i = 0; i < length; i++) {
                                        j = i;
                                        while (j > 0 && arr[j] < arr[j - 1]) {
                                          temp = arr[j]
                                          arr[j] = arr[j - 1];
                                          arr[j - 1] = temp;
                                          j--;
                                        }
                                      }

                            function merge(left, right) {
  var result = [],
    lLen = left.length,
    rLen = right.length,
    l = 0,
    r = 0;
    (l < lLen && r < rLen) {
    if (left[l] < right[r]) {
      result.push(left[l++]);
    } else {
      result.push(right[r++]);
    }
  }
  //remaining part needs to be addred to the result
  return result.concat(left.slice(l)).concat(right.slice(r));
}
